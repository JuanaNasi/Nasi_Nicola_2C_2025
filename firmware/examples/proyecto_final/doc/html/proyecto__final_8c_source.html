150void Vibrar(float pitch, float roll){
151
152 float desviacion_pitch = fabs(pitch - pitch_ref);
153 float desviacion_roll = fabs(roll - roll_ref);
154
155 if (desviacion_pitch > 10 || desviacion_roll > 7){
156 cont++;
157 }
158
159 if (cont > 20){
160 L293SetSpeed(MOTOR_2, 90);
161 vTaskDelay(3000 / portTICK_PERIOD_MS); // Durante este tiempo no mide, tiempo al usuario para enderezarse
162 L293SetSpeed(MOTOR_2, 0);
163 cont = 0;
164 }
165}
